function getRandomColor() {
    var letters = '0123456789ABCDEF';
    var color = '#';
    for (var i = 0; i < 6; i++) {
        color += letters[Math.floor(Math.random() * 16)];
    }
    return color;
}

function loadOptions() {
    lava.loadOptions('MyFancyChart', {
        colors: [getRandomColor()]
    });
}

var chartJson = {
    label: 'MyFancyChart',
    type: 'ScatterChart',
    elementId: 'chart_div',
    datatable: [
        ['Age', 'Cash'],
        [12,  101],
        [15,  240],
        [18,  280],
        [32,  324]
    ],
    options: {
        colors: ['green'],
        chartArea: {width: '50%'}
    }
};

lava.store(chartJson);
lava.run();
